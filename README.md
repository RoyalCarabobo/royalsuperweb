-- Tabla de Usuarios (Vendedores y Admins)
CREATE TABLE usuarios (
    id uuid REFERENCES auth.users NOT NULL PRIMARY KEY,

    correo TEXT UNIQUE NOT NULL,
    rol TEXT CHECK (rol IN ('admin', 'vendedor')) DEFAULT 'vendedor',
    status TEXT CHECK (status IN ('habilitado', 'deshabilitado')) DEFAULT 'habilitado',
    fecha_creacion TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- Tabla de Clientes
CREATE TABLE clientes (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    razon_social TEXT NOT NULL,
    rif TEXT UNIQUE NOT NULL,
    encargado TEXT NOT NULL,
    vendedor_id uuid REFERENCES usuarios(id), 
    direccion TEXT NOT NULL,
    telefono TEXT NOT NULL,
    email TEXT NOT NULL,
    foto_rif_url TEXT NOT NULL,
    foto_fachada_url TEXT NOT NULL,
    status TEXT CHECK (status IN ('pendiente', 'habilitado', 'deshabilitado')) DEFAULT 'pendiente',
    fecha_registro TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL
);

-- Tabla de Productos (Royal Super Oil)
CREATE TABLE productos (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    nombre TEXT NOT NULL,
    descripcion TEXT NOT NULL,
    stock INTEGER DEFAULT 0,
    precio_contado DECIMAL(10,2) NOT NULL,
    precio_credito DECIMAL(10,2) NOT NULL,
    status TEXT CHECK (status IN ('habilitado', 'deshabilitado')) DEFAULT 'habilitado'
);

-- Creamos una secuencia que empieza en 1
CREATE SEQUENCE correlativo_pedidos_seq START WITH 1;

-- Tabla de Pedidos (Cabecera)
CREATE TABLE pedidos (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    -- Usamos LPAD para rellenar con ceros a la izquierda hasta 2 dígitos (o los que necesites)
    numero_control TEXT UNIQUE DEFAULT LPAD(nextval('correlativo_pedidos_seq')::text, 2, '0'),
     cliente_id BIGINT REFERENCES clientes(id) NOT NULL,
    vendedor_id uuid REFERENCES usuarios(id) NOT NULL,
    monto_total DECIMAL(10,2) DEFAULT 0,
    -- Estados de flujo: proceso logístico
    status_logistico TEXT CHECK (status_logistico IN ('pendiente', 'aprobado', 'despachado', 'entregado')) DEFAULT 'pendiente',
    -- Estados de pago: flujo financiero
    status_pago TEXT CHECK (status_pago IN ('por cobrar', 'pagado', 'anulado')) DEFAULT 'por cobrar',
    dias_credito INTEGER DEFAULT 0,
    fecha_pedido TIMESTAMP WITH TIME ZONE DEFAULT timezone('utc'::text, now()) NOT NULL,
    fecha_vencimiento TIMESTAMP WITH TIME ZONE -- Se calcula sumando dias_credito
);
   

-- Tabla de Detalles del Pedido (Productos vendidos)
CREATE TABLE detalles_pedido (
    id BIGINT GENERATED BY DEFAULT AS IDENTITY PRIMARY KEY,
    pedido_id BIGINT REFERENCES pedidos(id) ON DELETE CASCADE,
    producto_id BIGINT REFERENCES productos(id),
    cantidad INTEGER NOT NULL,
    precio_historico DECIMAL(10,2) NOT NULL, -- Precio al momento de la venta
    subtotal DECIMAL(10,2) GENERATED ALWAYS AS (cantidad * precio_historico) STORED
);

-- Función para calcular fecha de vencimiento automáticamente
CREATE OR REPLACE FUNCTION calcular_vencimiento()
RETURNS TRIGGER AS $$
BEGIN
    NEW.fecha_vencimiento := NEW.fecha_pedido + (NEW.dias_credito || ' days')::interval;
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER tr_calcular_vencimiento
BEFORE INSERT OR UPDATE OF dias_credito ON pedidos
FOR EACH ROW EXECUTE FUNCTION calcular_vencimiento();

-- ==========================================================
-- 4. VISTA DE CARTERA Y MOROSIDAD
-- Esta vista te permite saber quién debe y quién está vencido sin queries locas.
-- ==========================================================
CREATE VIEW vista_cartera_clientes AS
SELECT 
    c.id AS cliente_id,
    c.razon_social,
    c.vendedor_id,
    COUNT(p.id) FILTER (WHERE p.status_pago = 'por cobrar') AS pedidos_pendientes,
    SUM(p.monto_total) FILTER (WHERE p.status_pago = 'por cobrar') AS deuda_total,
    COUNT(p.id) FILTER (WHERE p.status_pago = 'por cobrar' AND p.fecha_vencimiento < now()) AS pedidos_vencidos
FROM clientes c
LEFT JOIN pedidos p ON c.id = p.cliente_id
GROUP BY c.id, c.razon_social, c.vendedor_id;

CREATE OR REPLACE FUNCTION actualizar_stock_por_pedido()
RETURNS TRIGGER AS $$
BEGIN
  UPDATE productos
  SET stock = stock - NEW.cantidad
  WHERE id = NEW.producto_id;
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER tr_descontar_stock
AFTER INSERT ON pedidos_detalles
FOR EACH ROW
EXECUTE FUNCTION actualizar_stock_por_pedido();

CREATE OR REPLACE FUNCTION revisar_pago_completo()
RETURNS TRIGGER AS $$
DECLARE
  v_total_pedido NUMERIC;
  v_total_abonado NUMERIC;
BEGIN
  -- 1. Obtener el monto total del pedido original
  SELECT monto_total INTO v_total_pedido FROM pedidos WHERE id = NEW.pedido_id;

  -- 2. Sumar todos los pagos CONFIRMADOS para este pedido
  SELECT COALESCE(SUM(monto_pagado), 0) INTO v_total_abonado 
  FROM pagos_reportados 
  WHERE pedido_id = NEW.pedido_id AND estatus = 'confirmado';

  -- 3. Actualizar el estatus en la tabla pedidos
  IF v_total_abonado >= v_total_pedido THEN
    UPDATE pedidos SET status_pago = 'pagado' WHERE id = NEW.pedido_id;
  ELSIF v_total_abonado > 0 THEN
    UPDATE pedidos SET status_pago = 'abono parcial' WHERE id = NEW.pedido_id;
  END IF;

  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

-- Trigger que se activa cuando el Admin cambia el estatus a 'confirmado'
CREATE TRIGGER tr_verificar_pago_total
AFTER UPDATE OF estatus ON pagos_reportados
FOR EACH ROW
WHEN (NEW.estatus = 'confirmado')
EXECUTE FUNCTION revisar_pago_completo();
